GET clusters
----

**Example**


    GET api/v1/clusters/


    200 OK
    {
      "href" : "http://your.ambari.server/api/v1/clusters/",
      "items" : [
        {
          "href" : "http://your.ambari.server/api/v1/clusters/c1",
          "Clusters" : {
            "cluster_name" : "c1",
            "version" : "HDP-1.3.0"
          }
        }
      ]
    }

**Schema**

    {
      "type":"object",
      "$schema": "http://json-schema.org/draft-03/schema",
      "title": "Clusters",
      "required":true,
      "properties":{
        "href": {
          "type":"string",
          "description": "This clusters API href.",
          "required":true
        },
        "items": {
          "type":"array",
          "title": "Cluster set",
          "required":true,
          "items":
          {
            "type":"object",
            "title": "Cluster",
            "description": "A Hadoop cluster.",
            "required":false,
            "properties":{
              "Clusters": {
                "type":"object",
                "title": "ClusterInfo",
                "description": "Cluster information.",
                "required":true,
                "properties":{
                  "cluster_name": {
                    "type":"string",
                    "title": "ClusterName",
                    "description": "The cluster name.",
                    "required":true
                  },
                  "version": {
                    "type":"string",
                    "title": "Version",
                    "description": "The stack version.",
                    "required":true
                  }
                }
              },
              "href": {
                "type":"string",
                "description": "The cluster API href.",
                "required":true
              }
            }
          }
        }
      }
    }
